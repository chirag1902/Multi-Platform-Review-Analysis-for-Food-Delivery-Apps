def plot_source(selected_source):
    df_filtered = df_combined[df_combined['data_source'] == selected_source]
    target_apps = ['Ubereats', 'Doordash', 'Grubhub']
    colors = ['green', 'red', 'orange']
    counts = df_filtered['app_name'].value_counts().to_dict()
    x = [counts.get(app, 0) for app in target_apps]

    fig = go.Figure(go.Bar(
        x=x, y=target_apps, orientation='h',
        marker_color=colors, text=x, textposition='auto'
    ))

    fig.update_layout(
        title=f"Review Counts in {selected_source}",
        xaxis_title='Number of Reviews',
        yaxis_title='App',
        height=400,
        plot_bgcolor='white'
    )
    return fig

def plot_app(selected_app):
    df_filtered = df_combined[df_combined['app_name'] == selected_app]
    target_sources = ['Reddit', 'Google Play', 'App Store']
    colors = ['red', 'green', 'blue']
    counts = df_filtered['data_source'].value_counts().to_dict()
    x = [counts.get(src, 0) for src in target_sources]

    fig = go.Figure(go.Bar(
        x=x, y=target_sources, orientation='h',
        marker_color=colors, text=x, textposition='auto'
    ))

    fig.update_layout(
        title=f"{selected_app} Review Counts by Data Source",
        xaxis_title='Number of Reviews',
        yaxis_title='Data Source',
        height=400,
        plot_bgcolor='white'
    )
    return fig

data_source_dropdown = widgets.Dropdown(
    options=sorted(df_combined['data_source'].unique()),
    description='Data Source:',
    style={'description_width': 'initial'}
)

app_name_dropdown = widgets.Dropdown(
    options=sorted(df_combined['app_name'].unique()),
    description='App Name:',
    style={'description_width': 'initial'}
)

# --------------------------
# Create output areas
# --------------------------

output_source = widgets.Output()
output_app = widgets.Output()

# --------------------------
# Widget change handlers
# --------------------------

def on_source_change(change):
    with output_source:
        clear_output(wait=True)
        fig = plot_source(change.new)
        fig.show()

def on_app_change(change):
    with output_app:
        clear_output(wait=True)
        fig = plot_app(change.new)
        fig.show()

data_source_dropdown.observe(on_source_change, names='value')
app_name_dropdown.observe(on_app_change, names='value')

# --------------------------
# Display widgets + outputs
# --------------------------

display(data_source_dropdown, output_source)
display(app_name_dropdown, output_app)

# Show initial plots
with output_source:
    fig = plot_source(data_source_dropdown.value)
    fig.show()

with output_app:
    fig = plot_app(app_name_dropdown.value)
    fig.show()

This dashboard presents a side-by-side comparison of review volumes by app and by data source.

The left chart displays the total number of reviews submitted for each food delivery app (UberEats, DoorDash, GrubHub) within the selected data source (e.g., Reddit).

The right chart highlights how many reviews a selected app has received across multiple platforms (App Store, Google Play, Reddit).

This dual-panel visualization helps assess both platform-specific popularity and app-level customer engagement, offering insights into where users are most vocal and which platforms generate the most feedback.
